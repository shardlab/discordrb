def initialize(id: nil, color: nil, colour: nil, spoiler: false)
  @id = id
  @spoiler = spoiler
  @components = []
  self.colour = (colour || color)

  yield self if block_given?
end

def colour=(colour)
  @colour = case colour
            when Array
              (colour[0] << 16) | (colour[1] << 8) | colour[2]
            when String
              colour.delete('#').to_i(16)
            else
              colour&.to_i
            end
end
