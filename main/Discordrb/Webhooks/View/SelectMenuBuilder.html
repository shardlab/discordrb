def option(label:, value:, description: nil, emoji: nil, default: nil)
  emoji = case emoji
          when Integer, String
            emoji.to_i.positive? ? { id: emoji } : { name: emoji }
          else
            emoji&.to_h
          end

  @options << { label: label, value: value, description: description, emoji: emoji, default: default }
end
