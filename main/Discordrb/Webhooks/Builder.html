Convenience method to add an embed using a block-style builder pattern

def add_embed(embed = nil)
  embed ||= Embed.new
  yield(embed)
  self << embed
  embed
end

def to_json_hash
  {
    content: @content,
    username: @username,
    avatar_url: @avatar_url,
    tts: @tts,
    embeds: @embeds.map(&:to_hash),
    allowed_mentions: @allowed_mentions&.to_hash
  }
end
