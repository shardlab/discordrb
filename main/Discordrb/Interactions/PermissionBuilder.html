def allow(object)
  case object
  when Discordrb::User, Discordrb::Member
    create_entry(object.id, USER, true)
  when Discordrb::Role
    create_entry(object.id, ROLE, true)
  else
    raise ArgumentError, "Unable to create permission for unknown type: #{object.class}"
  end
end
