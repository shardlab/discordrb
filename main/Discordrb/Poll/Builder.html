def initialize(question:, layout: :default, duration: 24, multiselect: false)

  @answers = []
  @question = question

def answer(text:, emoji: nil)
  emoji = Emoji.build_emoji_hash(emoji, prefix: false) if emoji

  @answers << { poll_media: { text: text, emoji: emoji }.compact }
end
