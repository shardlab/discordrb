def accept(token, invite_code)
  Discordrb::API.request(
    :invite_code,
    nil,
    :post,
    "#{Discordrb::API.api_base}/invites/#{invite_code}",
    nil,
    Authorization: token
  )
end

def delete(token, code, reason = nil)
  Discordrb::API.request(
    :invites_code,
    nil,
    :delete,
    "#{Discordrb::API.api_base}/invites/#{code}",
    Authorization: token,
    'X-Audit-Log-Reason': reason
  )
end

def resolve(token, invite_code, counts = true)
  Discordrb::API.request(
    :invite_code,
    nil,
    :get,
    "#{Discordrb::API.api_base}/invites/#{invite_code}#{'?with_counts=true' if counts}",
    Authorization: token
  )
end
