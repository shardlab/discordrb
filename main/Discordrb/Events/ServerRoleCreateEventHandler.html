def matches?(event)
  # Check for the proper event type
  return false unless event.is_a? ServerRoleCreateEvent

  [
    matches_all(@attributes[:name], event.name) do |a, e|
      a == if a.is_a? String
             e.to_s
           else
             e
           end
    end
  ].reduce(true, &:&)
end
