def matches?(event)
  # Check for the proper event type
  return false unless event.is_a? UserBanEvent

  [
    matches_all(@attributes[:user], event.user) do |a, e|
      case a
      when String
        a == e.name
      when Integer
        a == e.id
      when :bot
        e.current_bot?
      else
        a == e
      end
    end,
    matches_all(@attributes[:server], event.server) do |a, e|
      a == case a
           when String
             e.name
           when Integer
             e.id
           else
             e
           end
    end
  ].reduce(true, &:&)
end
